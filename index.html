// ============================================================ // SHIROTECH BILLING API - WITH CLEAR ALL CLIENTS // ============================================================ function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { try { var action = e && e.parameter ? e.parameter.action : null; var data = e && e.parameter && e.parameter.data ? JSON.parse(e.parameter.data) : null; if (!action) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Missing action parameter' })).setMimeType(ContentService.MimeType.JSON); } switch(action) { case 'getClients': return getClients(); case 'addClient': return addClient(data); case 'updateClient': return updateClient(data); case 'deleteClient': return deleteClient(data ? data.id : null); case 'clearAllClients': return clearAllClients(); case 'getPayments': return getPayments(data ? data.clientId : null); case 'addPayment': return addPayment(data); case 'getTickets': return getTickets(); case 'addTicket': return addTicket(data); case 'updateTicket': return updateTicket(data); case 'getDashboard': return getDashboard(); default: return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Unknown action: ' + action })).setMimeType(ContentService.MimeType.JSON); } } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } function getSheet(name) { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name); } // ----- GET CLIENTS ----- function getClients() { try { var sheet = getSheet('CLIENTS'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Sheet "CLIENTS" not found' })).setMimeType(ContentService.MimeType.JSON); } var data = sheet.getDataRange().getValues(); if (data.length < 2) { return ContentService.createTextOutput(JSON.stringify({ success: true, data: [] })).setMimeType(ContentService.MimeType.JSON); } var headers = data[0]; var clients = []; for (var i = 1; i < data.length; i++) { var row = data[i]; // Skip rows where Name is empty if (!row[1] || row[1].toString().trim() === '') continue; var client = {}; for (var j = 0; j < headers.length; j++) { client[headers[j]] = row[j]; } clients.push(client); } return ContentService.createTextOutput(JSON.stringify({ success: true, data: clients })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ----- ADD CLIENT ----- function addClient(data) { try { var sheet = getSheet('CLIENTS'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Sheet "CLIENTS" not found' })).setMimeType(ContentService.MimeType.JSON); } var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; if (headers.indexOf('DueDate') === -1 && headers.indexOf('Due Date') === -1) { sheet.insertColumnAfter(7); sheet.getRange(1, 8).setValue('DueDate'); } var colCount = sheet.getLastColumn(); var row = []; for (var i = 0; i < colCount; i++) { row.push(''); } row[0] = data.id || (sheet.getLastRow()); row[1] = data.name || ''; row[2] = data.contact || ''; row[3] = data.address || ''; row[4] = data.plan || ''; row[5] = data.amount || 0; row[6] = data.installDate || ''; row[7] = data.dueDate || ''; row[8] = data.accountName || ''; row[9] = data.napbox || ''; row[10] = data.agent || ''; row[11] = data.status || 'Active'; row[12] = data.statusLog ? JSON.stringify(data.statusLog) : ''; sheet.appendRow(row); return ContentService.createTextOutput(JSON.stringify({ success: true, message: 'Client added successfully' })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ----- UPDATE CLIENT ----- function updateClient(data) { try { var sheet = getSheet('CLIENTS'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Sheet "CLIENTS" not found' })).setMimeType(ContentService.MimeType.JSON); } var rows = sheet.getDataRange().getValues(); var headers = rows[0]; var colCount = headers.length; for (var i = 1; i < rows.length; i++) { if (rows[i][0] == data.id) { var row = []; for (var j = 0; j < colCount; j++) { row.push(''); } row[0] = data.id; row[1] = data.name || ''; row[2] = data.contact || ''; row[3] = data.address || ''; row[4] = data.plan || ''; row[5] = data.amount || 0; row[6] = data.installDate || ''; row[7] = data.dueDate || ''; row[8] = data.accountName || ''; row[9] = data.napbox || ''; row[10] = data.agent || ''; row[11] = data.status || 'Active'; row[12] = data.statusLog ? JSON.stringify(data.statusLog) : ''; sheet.getRange(i+1, 1, 1, colCount).setValues([row]); return ContentService.createTextOutput(JSON.stringify({ success: true, message: 'Client updated successfully' })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Client not found' })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ----- DELETE CLIENT ----- function deleteClient(id) { try { var sheet = getSheet('CLIENTS'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Sheet "CLIENTS" not found' })).setMimeType(ContentService.MimeType.JSON); } var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][0] == id) { sheet.deleteRow(i+1); return ContentService.createTextOutput(JSON.stringify({ success: true, message: 'Client deleted successfully' })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Client not found' })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ----- CLEAR ALL CLIENTS ----- function clearAllClients() { try { var sheet = getSheet('CLIENTS'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Sheet "CLIENTS" not found' })).setMimeType(ContentService.MimeType.JSON); } var data = sheet.getDataRange().getValues(); if (data.length <= 1) { return ContentService.createTextOutput(JSON.stringify({ success: true, message: 'No clients to delete' })).setMimeType(ContentService.MimeType.JSON); } // Delete all rows except header (row 1) var totalRows = data.length; if (totalRows > 1) { sheet.deleteRows(2, totalRows - 1); } // Reset the nextClientId counter in the sheet? We'll handle this in the HTML. return ContentService.createTextOutput(JSON.stringify({ success: true, message: 'All clients deleted successfully' })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ----- GET PAYMENTS ----- function getPayments(clientId) { try { var sheet = getSheet('PAYMENTS'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Sheet "PAYMENTS" not found' })).setMimeType(ContentService.MimeType.JSON); } var data = sheet.getDataRange().getValues(); if (data.length < 2) { return ContentService.createTextOutput(JSON.stringify({ success: true, data: [] })).setMimeType(ContentService.MimeType.JSON); } var headers = data[0]; var payments = []; for (var i = 1; i < data.length; i++) { if (clientId && data[i][0] != clientId) continue; var row = data[i]; var payment = {}; for (var j = 0; j < headers.length; j++) { payment[headers[j]] = row[j]; } payments.push(payment); } return ContentService.createTextOutput(JSON.stringify({ success: true, data: payments })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ----- ADD PAYMENT ----- function addPayment(data) { try { var sheet = getSheet('PAYMENTS'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Sheet "PAYMENTS" not found' })).setMimeType(ContentService.MimeType.JSON); } var row = [ data.clientId || '', data.amount || 0, data.date || new Date().toISOString().split('T')[0], data.period || '', data.orNumber || '', data.method || 'Cash' ]; sheet.appendRow(row); return ContentService.createTextOutput(JSON.stringify({ success: true, message: 'Payment recorded successfully' })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ----- GET TICKETS ----- function getTickets() { try { var sheet = getSheet('TICKETS'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Sheet "TICKETS" not found' })).setMimeType(ContentService.MimeType.JSON); } var data = sheet.getDataRange().getValues(); if (data.length < 2) { return ContentService.createTextOutput(JSON.stringify({ success: true, data: [] })).setMimeType(ContentService.MimeType.JSON); } var headers = data[0]; var tickets = []; for (var i = 1; i < data.length; i++) { var row = data[i]; var ticket = {}; for (var j = 0; j < headers.length; j++) { ticket[headers[j]] = row[j]; } tickets.push(ticket); } return ContentService.createTextOutput(JSON.stringify({ success: true, data: tickets })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ----- ADD TICKET ----- function addTicket(data) { try { var sheet = getSheet('TICKETS'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Sheet "TICKETS" not found' })).setMimeType(ContentService.MimeType.JSON); } var row = [ data.id || (sheet.getLastRow()), data.clientId || '', data.subject || '', data.description || '', data.priority || 'Medium', data.status || 'Open', data.createdAt || new Date().toISOString().split('T')[0], data.updates ? JSON.stringify(data.updates) : '' ]; sheet.appendRow(row); return ContentService.createTextOutput(JSON.stringify({ success: true, message: 'Ticket created successfully' })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ----- UPDATE TICKET ----- function updateTicket(data) { try { var sheet = getSheet('TICKETS'); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Sheet "TICKETS" not found' })).setMimeType(ContentService.MimeType.JSON); } var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][0] == data.id) { var row = [ data.id, data.clientId || '', data.subject || '', data.description || '', data.priority || 'Medium', data.status || 'Open', data.createdAt || '', data.updates ? JSON.stringify(data.updates) : '' ]; sheet.getRange(i+1, 1, 1, row.length).setValues([row]); return ContentService.createTextOutput(JSON.stringify({ success: true, message: 'Ticket updated successfully' })).setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput(JSON.stringify({ success: false, error: 'Ticket not found' })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ----- GET DASHBOARD ----- function getDashboard() { try { var clientsResult = getClients(); var clientsData = JSON.parse(clientsResult.getContent()); var clients = clientsData.success ? clientsData.data : []; var paymentsResult = getPayments(null); var paymentsData = JSON.parse(paymentsResult.getContent()); var payments = paymentsData.success ? paymentsData.data : []; var totalClients = clients.length; var activeClients = 0; var totalCollectible = 0; for (var i = 0; i < clients.length; i++) { if (clients[i].Status === 'Active') { activeClients++; totalCollectible += parseFloat(clients[i].Amount) || 0; } } var totalCollected = 0; for (var j = 0; j < payments.length; j++) { totalCollected += parseFloat(payments[j].Amount) || 0; } return ContentService.createTextOutput(JSON.stringify({ success: true, data: { totalClients: totalClients, activeClients: activeClients, totalCollectible: totalCollectible, totalCollected: totalCollected } })).setMimeType(ContentService.MimeType.JSON); } catch(error) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: error.toString() })).setMimeType(ContentService.MimeType.JSON); } }
